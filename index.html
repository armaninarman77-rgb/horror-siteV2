<script>
let time=0,fearLevel=0;
const timerEl=document.getElementById('time');
const screen=document.getElementById('screen');
const screamAudio=document.getElementById('screamAudio');
const heartAudio=document.getElementById('heartbeat');
const screamers=['screamer1','screamer2','screamer3'].map(id=>document.getElementById(id));

const messages=['Ты слышишь это?','Оно за твоей спиной','Не оборачивайся','Здесь кто-то есть...','Беги!','Кто-то наблюдает...'];
const diary=['День 1: Я видел это в зеркале...','День 3: Звуки из подвала...','День 7: Оно знает моё имя...','День 12: Я больше не один...','День 20: Оно приближается...'];
let diaryIndex=0;

function init(){
setInterval(()=>{time++;timerEl.textContent=time;updateFearLevel();},1000);
setInterval(showRandomMessage,Math.random()*30000+15000);
setInterval(addArtifact,Math.random()*20000+10000);
setInterval(dropBlood,Math.random()*10000+5000);
setInterval(createDeadPixel,Math.random()*8000+3000);
setInterval(showSilhouette,Math.random()*40000+20000);
try{heartAudio.volume=.2;heartAudio.play();}catch(e){console.warn(e);}
history.pushState({},'', '/you-are-not-alone');
}

function updateFearLevel(){
if(fearLevel>70) document.body.style.filter=`blur(${fearLevel/50}px)`;
if(heartAudio) heartAudio.volume=Math.min(fearLevel/100,.8);
}

function showRandomMessage(){screen.textContent=messages[Math.floor(Math.random()*messages.length)];}
function addArtifact(){screen.textContent+=`\n[ERROR 404: SOUL_NOT_FOUND] ${Math.random().toString(16).substr(2,8)}`;}
function dropBlood(){const b=document.createElement('div');b.className='blood';b.style.left=Math.random()*window.innerWidth+'px';b.style.top='-10px';document.body.appendChild(b);setTimeout(()=>b.remove(),4000);}
function createDeadPixel(){const p=document.createElement('div');p.className='dead-pixel';p.style.left=Math.random()*window.innerWidth+'px';p.style.top=Math.random()*window.innerHeight+'px';document.body.appendChild(p);setTimeout(()=>p.remove(),5000);}
function showSilhouette(){const s=document.createElement('div');s.className='silhouette';s.style.setProperty('--top',Math.random()*80+'%');s.style.setProperty('--left',Math.random()*80+'%');document.body.appendChild(s);setTimeout(()=>s.remove(),300);}
function triggerScreamer(s){s.style.top=Math.random()*80+'%';s.style.left=Math.random()*80+'%';s.classList.add('active');try{screamAudio.currentTime=0;screamAudio.play();}catch(e){console.warn(e);}fearLevel+=25;showDiaryEntry();setTimeout(()=>s.classList.remove('active'),1200);}
function triggerScreamerRandom(){triggerScreamer(screamers[Math.floor(Math.random()*screamers.length)]);}
function safeAction(){alert('Вы пережили этот момент... пока');fearLevel=Math.max(fearLevel-10,0);screen.textContent='Кажется, опасность миновала...';}
function fakeExplosion(){alert('Система заблокирована...');document.body.style.background='#300';document.body.style.filter='sepia(1) saturate(3) hue-rotate(180deg)';setTimeout(()=>alert('ПОЗДНО'),3000);}
function showDiaryEntry(){if(diaryIndex<diary.length)screen.textContent+=`\n\n${diary[diaryIndex++]}`;}

let secretCode=['ArrowUp','ArrowUp','ArrowDown','ArrowDown','ArrowLeft','ArrowRight','ArrowLeft','ArrowRight','b','a'],codeInput=[];
document.addEventListener('keydown',e=>{codeInput.push(e.key);if(codeInput.length>secretCode.length)codeInput.shift();if(codeInput.join(',')===secretCode.join(',')){triggerScreamer(screamers[0]);codeInput=[];alert('Вы разбудили нечто...');}});

window.onload=init;
</script>
